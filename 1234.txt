  ctx.fillRect(0, 0, W, H);

  // Grid lines
  ctx.strokeStyle = 'rgba(255,255,255,0.04)';
  ctx.lineWidth = 1;
  for (let x = 0; x < W; x += 40) {
    ctx.beginPath(); ctx.moveTo(x, 0); ctx.lineTo(x, H); ctx.stroke();
  }
  for (let y = 0; y < H; y += 40) {
    ctx.beginPath(); ctx.moveTo(0, y); ctx.lineTo(W, y); ctx.stroke();
  }

  // Ashoka Chakra (stylised)
  const cx = W * 0.5, cy = H * 0.5, R = 90;
  ctx.save();
  ctx.translate(cx, cy);

  // Outer ring
  ctx.beginPath();
  ctx.arc(0, 0, R, 0, Math.PI * 2);
  ctx.strokeStyle = 'rgba(20,80,180,0.5)';
  ctx.lineWidth = 2;
  ctx.stroke();

  // 24 spokes
  for (let i = 0; i < 24; i++) {
    const angle = (i / 24) * Math.PI * 2 - Math.PI / 2;
    ctx.save();
    ctx.rotate(angle);
    const grad = ctx.createLinearGradient(0, 0, 0, R - 10);
    grad.addColorStop(0, 'rgba(255,107,0,0.8)');
    grad.addColorStop(1, 'rgba(26,58,107,0.2)');
    ctx.strokeStyle = grad;
    ctx.lineWidth = 1.5;
    ctx.beginPath();
    ctx.moveTo(0, 8);
    ctx.lineTo(0, R - 10);
    ctx.stroke();
    ctx.restore();
  }

  // Inner circle
  ctx.beginPath();
  ctx.arc(0, 0, 18, 0, Math.PI * 2);
  const innerGrad = ctx.createRadialGradient(0, 0, 4, 0, 0, 18);
  innerGrad.addColorStop(0, '#FF6B00');
  innerGrad.addColorStop(1, '#8B3500');
  ctx.fillStyle = innerGrad;
  ctx.fill();

  ctx.restore();

  // NODE ORBITS — 7 patent use case nodes orbiting
  const nodes = [
    { label: 'AI\nInventor', angle: -Math.PI / 2, color: '#FF6B00' },
    { label: 'Drafting\nSec.10', angle: -Math.PI / 2 + (Math.PI * 2 / 7), color: '#D4A017' },
    { label: 'Prior Art\nTKDL', angle: -Math.PI / 2 + (Math.PI * 4 / 7), color: '#0A5C36' },
    { label: 'FER\nReply', angle: -Math.PI / 2 + (Math.PI * 6 / 7), color: '#1A3A6B' },
    { label: 'Litigation\nDHC', angle: -Math.PI / 2 + (Math.PI * 8 / 7), color: '#FF6B00' },
    { label: 'PLI\nStrategy', angle: -Math.PI / 2 + (Math.PI * 10 / 7), color: '#D4A017' },
    { label: 'PCT\nGlobal', angle: -Math.PI / 2 + (Math.PI * 12 / 7), color: '#0A5C36' },
  ];

  const orbitR = R + 60;
  nodes.forEach(n => {
    const nx = cx + Math.cos(n.angle) * orbitR;
    const ny = cy + Math.sin(n.angle) * orbitR;

    // Connection line
    ctx.beginPath();
    ctx.moveTo(cx, cy);
    ctx.lineTo(nx, ny);
    ctx.strokeStyle = n.color + '44';
    ctx.lineWidth = 1;
    ctx.setLineDash([4, 6]);
    ctx.stroke();
    ctx.setLineDash([]);

    // Node circle
    ctx.beginPath();
    ctx.arc(nx, ny, 22, 0, Math.PI * 2);
    ctx.fillStyle = n.color + '22';
    ctx.fill();
    ctx.strokeStyle = n.color;
    ctx.lineWidth = 1.5;
    ctx.stroke();

    // Node text
    const lines = n.label.split('\n');
    ctx.fillStyle = '#fff';
    ctx.font = `bold 8px 'JetBrains Mono', monospace`;
    ctx.textAlign = 'center';
    ctx.textBaseline = 'middle';
    lines.forEach((line, i) => {
      const offset = lines.length === 2 ? (i - 0.5) * 10 : 0;
      ctx.fillText(line, nx, ny + offset);
    });
  });

  // TITLE TEXT (left)
  ctx.textAlign = 'left';
  ctx.fillStyle = 'rgba(255,255,255,0.9)';
  ctx.font = `bold 13px 'JetBrains Mono', monospace`;
  ctx.fillText('AI × INDIAN', 18, 36);
  ctx.fillStyle = '#FF6B00';
  ctx.font = `bold 15px 'JetBrains Mono', monospace`;
  ctx.fillText('PATENT LAW', 18, 56);
  ctx.fillStyle = 'rgba(255,255,255,0.35)';
  ctx.font = `10px 'JetBrains Mono', monospace`;
  ctx.fillText('ECOSYSTEM 2026', 18, 74);

  // TRICOLOR BAR at bottom
  const bh = 5;
  ctx.fillStyle = '#FF6B00';
  ctx.fillRect(0, H - bh * 3, W, bh);
  ctx.fillStyle = '#fff';
  ctx.fillRect(0, H - bh * 2, W, bh);
  ctx.fillStyle = '#0A5C36';
  ctx.fillRect(0, H - bh, W, bh);

  // Stats on right side
  const stats = [
    ['3,30,000+', 'Applications FY25'],
    ['3.6L', 'TKDL Formulations'],
    ['2,444', 'PCT Filings (2024)'],
  ];
  ctx.textAlign = 'right';
  stats.forEach((s, i) => {
    const sy = H - 80 + i * 26;
    ctx.fillStyle = '#FF6B00';
    ctx.font = `bold 14px 'JetBrains Mono', monospace`;
    ctx.fillText(s[0], W - 18, sy);
    ctx.fillStyle = 'rgba(255,255,255,.4)';
    ctx.font = `9px 'JetBrains Mono', monospace`;
    ctx.fillText(s[1], W - 18, sy + 13);
  });
}
